# The list of input method IDs that we support. This filtering is necessary
# since some input methods are definitely unnecessary for us. For example, we
# should disable "ja:anthy", "zh:cangjie", and "zh:pinyin" engines in
# ibus-m17n since we (will) have better equivalents outside of ibus-m17n.
# The list will be processed by scripts.

# The order of the list is important. In Chrome's "Languages and Input" options
# page, we list input methods in this order, when more than one input methods
# are available for a language (ex. French).

# List items are formatted as follows.
#
# input_method_id     keyboard_overlay_id   # Comment
#
# input_method_id: The ID used internally by ibus.
# keyboard_overlay_id: The ID to specify which keyboard layout is used for the
#                      keyboard overlay dialog.
# Each list item defines the correspondence between input_method_id and
# keyboard_overlay_id.

# Dutch
xkb:nl::nld           nl           # Netherlands - Dutch
xkb:be::nld           nl           # Belgium - Dutch

# French
xkb:fr::fra           fr           # France - French
xkb:be::fra           fr           # Belgium - French
xkb:ca::fra           fr_CA        # Canada - French
xkb:ch:fr:fra         fr           # Switzerland - French

# German
xkb:de::ger           de           # Germany - German
xkb:be::ger           de           # Belgium - German
xkb:ch::ger           de           # Switzerland - German

# Japanese
mozc                  en_US        # ibus-mozc - Japanese (with US keyboard)
mozc-jp               ja           # ibus-mozc - Japanese (with JP keyboard)
mozc-dv               en_US_dvorak # ibus-mozc - Japanese (with Dvorak keyboard)
xkb:jp::jpn           ja           # Japan - Japanese

# Russian
xkb:ru::rus           ru           # Russia - Russian
xkb:ru:phonetic:rus   ru           # Russia - Phonetic - Russian

# Thai
m17n:th:kesmanee      th           # Thai (simulate the Kesmanee keyboard)
m17n:th:pattachote    th           # Thai (simulate the Pattachote keyboard)
m17n:th:tis820        th           # Thai (simulate the TIS-820.2538 keyboard)

# Traditional Chinese
chewing               zh_TW        # ibus-chewing - Traditional Chinese
m17n:zh:cangjie       zh_TW        # Traditional Chinese (Cangjie)
m17n:zh:quick         zh_TW        # Traditional Chinese (Quick) in m17n-contrib
# TODO(suzhe): Add CantonHK and Stroke5 if (it's really) necessary.

# Vietnamese

m17n:vi:telex         vi           # Vietnames (TELEX key sequence)
m17n:vi:viqr          vi           # Vietnames (VIQR key sequence)
m17n:vi:vni           vi           # Vietnames (VNI key sequence)

# feature.
